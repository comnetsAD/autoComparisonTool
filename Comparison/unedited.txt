var dfpTakeoverRunning = false;
function updateBackground() {
	var hfxtmfd = document.getElementById("hfxtmfd");
	var svglayer = document.getElementById("svglayer");
	if (dfpTakeoverRunning) {
		hfxtmfd.classList.add("fixed-width");
		var bounds = hfxtmfd.getBoundingClientRect();
		var svgTop = bounds.top;
		var svgLeft = bounds.left;
		var spots = ['filmstrip_1','filmstrip_2','filmstrip_3','sidekick_1','sidekick_2','sidekick_3','sidekick_mobile_1','super_leaderboard_1','super_leaderboard_2','super_leaderboard_3','ultra_leaderboard_1','ultra_leaderboard_2','wedge_1','wedge_2'];
		spots.forEach(function(spot) {
			var clip = document.getElementById("clip_dfp_"+spot);
			if (clip) {
				var div = document.getElementById("dfp_"+spot);
				var r = div ? div.getBoundingClientRect() :
					{left:0, top:0, width:0, height:0};

				clip.setAttribute("x", (r.left-svgLeft)+"px");
				clip.setAttribute("y", (r.top-svgTop)+"px");
				clip.setAttribute("width", r.width+"px");
				clip.setAttribute("height", r.height+"px");
			}
		});
	}
	svglayer.style.setProperty("display", dfpTakeoverRunning ? "initial" : "none");
	hfxtmfd.style.setProperty("background-color", dfpTakeoverRunning ? "transparent" : "white");
}

(function(window, document) {
	document.addEventListener('DOMContentLoaded', updateBackground);
	window.addEventListener('resize', updateBackground);
})(window, document);
</script>
